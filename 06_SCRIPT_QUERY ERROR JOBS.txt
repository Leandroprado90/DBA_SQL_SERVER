SELECT TOP (1000) [ID]
	,[DatabaseName]
	,[SchemaName]
	,[ObjectName]
	,[ObjectType]
	,[IndexName]
	,[IndeType]
	,[StatisticsName]
	,[PartitionNumber]
	,[ExtendedInfo]
	,[Command]
	,[CommandType]
	,[StartTime]
	,[EndTime]
	,[ErrorNumber]
	,[ErrorMessage]
FROM [master].[dbo].[CommandLog]
order by [ErrorNumber] desc
